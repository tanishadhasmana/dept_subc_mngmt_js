--------set up TS---------
install first using-
 npm install --save-dev typescript ts-node @types/node @types/express @types/cookie-parser @types/jsonwebtoken @types/bcrypt

then make tsconfig.json-
{
  "compilerOptions": {
    "target": "ES6",
    "module": "commonjs",
    "outDir": "./dist",
    "rootDir": "./",
    "strict": true,
    "esModuleInterop": true
  },
  "include": ["**/*.ts"],
  "exclude": ["node_modules"]
}
